## Usage in Code

```typescript
// Determine OSS status
const isOpenSource = (project: Project): boolean => {
  const githubUrl = project.githubUrl || '';

  // bitcoin-apps-suite = always OSS
  if (githubUrl.includes('bitcoin-apps-suite')) return true;

  // bitcoin-corp = OSS except website/corporation
  if (githubUrl.includes('bitcoin-corp')) {
    return !githubUrl.includes('website') && !githubUrl.includes('corporation');
  }

  // b0ase = never OSS (commercial)
  if (githubUrl.includes('b0ase')) return false;

  // Default to private for safety
  return false;
};
```

## Updating This Document

To refresh the public repo list:

```bash
# bitcoin-apps-suite public repos
curl -s "https://api.github.com/orgs/bitcoin-apps-suite/repos?per_page=100" | jq -r '.[].name'

# bitcoin-corp public repos
curl -s "https://api.github.com/orgs/bitcoin-corp/repos?per_page=100" | jq -r '.[].name'

# b0ase public repos (mostly forks/experiments)
curl -s "https://api.github.com/users/b0ase/repos?per_page=100" | jq -r '.[].name'
```
