## Custom Scraper Implementation (Claude Code)

### Spotify New Release Scraper

```javascript
// /Volumes/2026/Projects/b0ase.com/scripts/scrape-spotify-releases.js

import SpotifyWebApi from 'spotify-web-api-node';
import fs from 'fs';

const spotifyApi = new SpotifyWebApi({
  clientId: process.env.SPOTIFY_CLIENT_ID,
  clientSecret: process.env.SPOTIFY_CLIENT_SECRET
});

async function scrapeNewReleases() {
  // Authenticate
  const data = await spotifyApi.clientCredentialsGrant();
  spotifyApi.setAccessToken(data.body['access_token']);

  // Get new releases
  const releases = await spotifyApi.getNewReleases({ limit: 50, country: 'GB' });

  const prospects = [];

  for (const album of releases.body.albums.items) {
    const artist = album.artists[0];

    // Filter: Independent artists (followers < 50k = likely independent)
    const artistDetails = await spotifyApi.getArtist(artist.id);

    if (artistDetails.body.followers.total < 50000 && artistDetails.body.followers.total > 5000) {
      // Sweet spot: Has audience but not signed to major

      prospects.push({
        artistName: artist.name,
        albumName: album.name,
        releaseDate: album.release_date,
        followers: artistDetails.body.followers.total,
        spotifyUrl: artist.external_urls.spotify,
        genres: artistDetails.body.genres,
        signal: 'new_album_release',
        prospectScore: calculateScore(artistDetails.body)
      });
    }
  }

  // Save to file for email campaign
  fs.writeFileSync(
    './leads/spotify-new-releases.json',
    JSON.stringify(prospects, null, 2)
  );

  console.log(`Found ${prospects.length} independent artist prospects`);
  return prospects;
}

function calculateScore(artist) {
  // Score based on likelihood to convert
  let score = 0;

  if (artist.followers.total > 10000) score += 30;
  if (artist.popularity > 40) score += 20;
  if (artist.genres.includes('electronic') || artist.genres.includes('hip-hop')) score += 10;

  return score;
}

scrapeNewReleases();
```

### Crunchbase Funding Scraper

```javascript
// /Volumes/2026/Projects/b0ase.com/scripts/scrape-funding-rounds.js

import axios from 'axios';
import fs from 'fs';

async function scrapeFundingRounds() {
  const crunchbaseKey = process.env.CRUNCHBASE_API_KEY;

  // Get recent funding rounds
  const response = await axios.get(
    'https://api.crunchbase.com/api/v4/searches/funding_rounds',
    {
      headers: { 'X-cb-user-key': crunchbaseKey },
      params: {
        field_ids: ['identifier', 'announced_on', 'funding_type', 'money_raised'],
        order: [{ field_id: 'announced_on', sort_order: 'desc' }],
        limit: 100
      }
    }
  );

  const prospects = [];

  for (const round of response.data.entities) {
    // Filter: Seed to Series A, last 60 days
    const daysAgo = (Date.now() - new Date(round.properties.announced_on)) / (1000 * 60 * 60 * 24);

    if (daysAgo <= 60 && ['seed', 'series_a'].includes(round.properties.funding_type)) {
      // Get company details
      const companyId = round.properties.identifier.entity_def_id;
      const company = await axios.get(
        `https://api.crunchbase.com/api/v4/entities/organizations/${companyId}`,
        { headers: { 'X-cb-user-key': crunchbaseKey } }
      );

      prospects.push({
        companyName: company.data.properties.name,
        fundingAmount: round.properties.money_raised?.value_usd,
        fundingType: round.properties.funding_type,
        announcedDate: round.properties.announced_on,
        website: company.data.properties.website,
        description: company.data.properties.short_description,
        signal: 'recent_funding',
        prospectScore: 85 // High intent - just got money
      });
    }
  }

  fs.writeFileSync(
    './leads/recent-funding-rounds.json',
    JSON.stringify(prospects, null, 2)
  );

  console.log(`Found ${prospects.length} recently funded startups`);
  return prospects;
}

scrapeFundingRounds();
```

### Product Hunt Launch Scraper

```javascript
// /Volumes/2026/Projects/b0ase.com/scripts/scrape-product-hunt.js

import axios from 'axios';
import cheerio from 'cheerio';
import fs from 'fs';

async function scrapeProductHunt() {
  const prospects = [];

  // Scrape last 7 days of launches
  for (let daysAgo = 0; daysAgo < 7; daysAgo++) {
    const date = new Date();
    date.setDate(date.getDate() - daysAgo);
    const dateStr = date.toISOString().split('T')[0];

    const url = `https://www.producthunt.com/posts?date=${dateStr}`;
    const response = await axios.get(url);
    const $ = cheerio.load(response.data);

    // Parse product listings
    $('.postsList .post').each((i, elem) => {
      const productName = $(elem).find('.postItem-title').text().trim();
      const description = $(elem).find('.postItem-tagline').text().trim();
      const upvotes = parseInt($(elem).find('.postItem-voteCount').text());
      const productUrl = 'https://www.producthunt.com' + $(elem).find('a').attr('href');

      // Filter: 100+ upvotes = getting traction
      if (upvotes >= 100) {
        prospects.push({
          productName,
          description,
          upvotes,
          productUrl,
          launchDate: dateStr,
          signal: 'product_hunt_launch',
          prospectScore: Math.min(upvotes / 10, 100) // Up to 100 score
        });
      }
    });
  }

  fs.writeFileSync(
    './leads/product-hunt-launches.json',
    JSON.stringify(prospects, null, 2)
  );

  console.log(`Found ${prospects.length} Product Hunt launches`);
  return prospects;
}

scrapeProductHunt();
```
